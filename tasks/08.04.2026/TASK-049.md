## Новый `DoseData` интерфейс (целевая схема)

```typescript
export interface DayDose {
    dayLabel: string;           // "День 1" / "Дни 2-5" / "Далее"
    singleDoseMg: number | null;
    timesPerDay: number | null;
    routeOfAdmin: string | null;
    note?: string | null;       // необязательная метка (нагрузочная / поддерживающая...)
}

export interface DoseData {
    // Главные поля (сохраняются как сводные строки для совместимости)
    dosing: string;              // авто-генерируется из daySchedule или вводится вручную
    duration: string;            // "5-7 дней"

    // Форма и путь
    formId: string | null;
    formType: string | null;     // tablet | powder | solution | ... (из выбранной формы)
    routeOfAdmin: string | null;

    // Однодневная доза (legacy / simple mode)
    singleDoseMg?: number | null;
    timesPerDay?: number | null;

    // Многодневное расписание (приоритет над legacy)
    daySchedule?: DayDose[] | null;

    // Упаковка
    packagingDescription?: string | null;

    // Разведение (инъекции + порошок)
    dilution?: {
        enabled: boolean;
        // Порошок во флаконе
        powderVialMg?: number | null;        // мг в одном флаконе
        reconstitutionVolumeMl?: number | null; // объём растворителя при разведении
        // Готовый раствор (или результат разведения порошка)
        drugAmountMg?: number | null;
        diluentType?: DiluentType | null;
        diluentVolumeMl?: number | null;
        concentrationMgPerMl?: number | null;  // авторасчёт
        volumeToDrawMl?: number | null;        // авторасчёт
    } | null;
}
```
